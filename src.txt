<?php
require 'db.php';
$message = '';

// 处理表单提交 (插入新员工)
if ($_SERVER["REQUEST_METHOD"] == "POST") {
    $emp_id = trim($_POST['employee_id']);
    $first_name = trim($_POST['first_name']);
    $last_name = trim($_POST['last_name']);
    $email = trim($_POST['email_address']);

    // 表单验证: 确保必填字段不为空
    if (empty($emp_id) || empty($first_name) || empty($last_name)) {
        $message = "<p class='error'>错误: 员工ID、名字和姓氏为必填项。</p>";
    } else {
        try {
            $stmt = $pdo->prepare("INSERT INTO employees (employee_id, first_name, last_name, email_address) VALUES (?, ?, ?, ?)");
            $stmt->execute([$emp_id, $first_name, $last_name, $email]);
            $message = "<p class='success'>员工添加成功！</p>";
        } catch (PDOException $e) {
            $message = "<p class='error'>添加失败: " . $e->getMessage() . "</p>";
        }
    }
}

// 获取所有员工数据
$stmt = $pdo->query("SELECT * FROM employees");
$employees = $stmt->fetchAll();
?>

                    </thead>
                    <tbody>
                        <?php foreach ($employees as $emp): ?>
